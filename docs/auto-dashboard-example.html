Then, with a dataset and dashboard spec

We can Meliorate for a Dashboard

                            var parsedDataSet = $.makeArray($('.originalTable').Meliorator('parse-table')[0]);
    // so, assuming we already have a dashboard spec (say it was saved in a previous dasbhoard save mode, or something like that...)
    var dashboardSpec = {
        "title": "Dashboard-2016-10-10T12:49:47.395Z",
        "spec": [{
            "domain": "Number",
            "series": ["Number", "Square", "Cube"],
            "aggregation": "NONE",
            "render": "POWER-TABLE"
        }, {
            "domain": "Number",
            "series": ["Square"],
            "aggregation": "NONE",
            "render": "LINE"
        }, {
            "domain": "Number",
            "series": ["Cube"],
            "aggregation": "NONE",
            "render": "LINE"
        }, {
            "domain": "Number",
            "series": ["Square", "Cube"],
            "aggregation": "NONE",
            "render": "LINE"
        }],
        "timestamp": "2016-10-10T12:49:47.395Z"
    }
    // then, we tell Meliorator to render a dashboard for this data, based off of this spec, as such...
    $('#analytics').Meliorator('dashboard', {
        data: parsedDataSet,
        dashboardSpec: dashboardSpec
    });
